# config our webcam to output MJPG video at 800x600
v4l2-ctl --list-formats
v4l2-ctl --set-fmt-video=width=800,height=600,pixelformat=1
v4l2-ctl --all
# Capture 30 frame from the cam and save to output.raw
 ./capture -d /dev/video0 -c 30 -o > output.raw

 # convert the captured MJPG into a useable video stream
 ffmpeg -f mjpeg -i output.raw -vcodec copy output.avi
 
 # incase you get a 'select timeout' error, try this:
 rmmod uvcvideo
 modprobe uvcvideo nodrop=1 timeout=5000
 
 # installing xawtv
 git clone git://linuxtv.org/xawtv4.git
 autoconf
 ./configure
 make install
